//fired  EH der von einer Einheit die gefeuerte Kugel ersetzt.
//Idee: KI kriegt Thermobarische raketen, die sie häufig auf infantrie schiesst. thermobarische munition wird beim schuss durch normale RPG ersetzt.
player addEventHandler ["FiredMan", {
_proj = _this#6;
_pos = getPosWorld _proj;
_rpg = createVehicle ["R_PG7_F", _pos];
_rpg setPosWorld _pos;
_rpg setVectorDirAndUp [vectorDir _proj, vectorUp _proj];
_rpg setVelocityModelSpace [0, 300, 0];
_rpg setShotParents [_this#0, _this#0];
deleteVehicle _proj;
}]
